create table Person(
       person_id long primary key auto_increment not null,
       full_name varchar(255) not null,
       gender varchar(255),
       height double,
       birthday date,
       birth_place varchar(255),
       death_cause varchar(255),
       death_date date,
       photo_id long,
       foreign key(photo_id) references Image(image_id) on delete cascade
);

create table Movie(
      movie_id long primary key auto_increment not null,
      title varchar(200) not null,
      description text,
      running_time_in_minutes int,
      chart_rating double,
      year int,
      image_id long,
      foreign key(image_id) references Image(image_id) on delete cascade
);

create table Image(
      image_id long primary key auto_increment not null,
      height int not null,
      width int not null,
      image_url text
);

create table `character`(
      character_id long primary key auto_increment not null,
      name varchar(255) not null,
      movie_id long not null,
      actor_id long not null,
      foreign key(movie_id) references Movie(movie_id) on delete cascade,
      foreign key(actor_id) references Person(person_id) on delete cascade
);

create table Genre(
      genre_id long primary key auto_increment not null,
      name varchar(255) not null
);

create table Movie_Writer(
     movie_id long not null,
     person_id long not null,
     primary key(movie_id, person_id)
);

create table Movie_Director(
   movie_id long not null,
   person_id long not null,
   primary key(movie_id, person_id)
);

create table Movie_Genre(
    movie_id long not null,
    genre_id long not null,
    primary key(movie_id, genre_id)
);

create table Movie_Actor(
    movie_id long not null,
    person_id long not null,
    primary key(movie_id, person_id)
);
